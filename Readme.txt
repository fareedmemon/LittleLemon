Dependencies:

pip install django djangorestframework djoser mysqlclient

Paths:

'' -> Users
'restaurant/' -> HTML response
'api/booking/tables' -> Browsing, creating and editing table bookings
'api/menu-item/' -> Browsing and creating menu-items
'api/menu-item/<int:pk>' -> Details, updating and deleting a single menu-item
'api/auth/users/' -> Browsing and creating users
'api/auth/token/login' -> Getting user token
